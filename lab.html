EX 05 – Certificate DApp
pragma solidity ^0.8.20;contract Certificate {
address public admin = msg.sender;

struct Cert {
    string name;
    string course;
    string grade;
    string date;
}

mapping(uint => Cert) public certs;

function issue(uint id, string memory n, string memory c, string memory g, string memory d) public {
    require(msg.sender == admin);
    certs[id] = Cert(n,c,g,d);
}

function get(uint id) public view returns (string memory,string memory,string memory,string memory) {
    Cert memory c = certs[id];
    return (c.name,c.course,c.grade,c.date);
}

}
EX 06 – Clinical Trial
pragma solidity ^0.8.0;contract ClinicalTrial {
struct Participant {
string name;
string data;
}

mapping(uint => Participant) public p;

function enroll(uint id, string memory name) public {
    p[id].name = name;
}

function record(uint id, string memory data) public {
    p[id].data = data;
}

function get(uint id) public view returns(string memory,string memory){
    return (p[id].name, p[id].data);
}

}

EX 08 – Smart Toll System
pragma solidity ^0.8.0;contract Toll {
mapping(address => uint) public balance;

function register() public {
    balance[msg.sender] = 100;
}

function pay(uint amt) public {
    require(balance[msg.sender] >= amt);
    balance[msg.sender] -= amt;
}

function getBalance() public view returns(uint){
    return balance[msg.sender];
}

}
