EX 04 – Hello World (Sawtooth)

Procedure: Install Sawtooth → Create JS processor → Run → Send transaction.

const { TransactionProcessor } = require('sawtooth-sdk/processor');
const crypto = require('crypto');
const FAMILY_NAME='hello_world';
const PREFIX=crypto.createHash('sha512').update(FAMILY_NAME).digest('hex').slice(0,6);const handler=(tx)=>{console.log(tx.payload.toString()); return Promise.resolve();};
const processor=new TransactionProcessor('tcp://localhost:4004');
processor.addHandler(handler,[PREFIX]);
processor.start();

EX 05 – Certificate DApp

Procedure: Write contract → Compile in Remix → Deploy → Use issue/get functions.

pragma solidity ^0.8.20;
contract Certificate{
 address admin=msg.sender;
 struct Cert{string name;string course;string grade;string date;}
 mapping(uint=>Cert) public certs;

 function issue(uint id,string memory n,string memory c,string memory g,string memory d) public{
  require(msg.sender==admin);
  certs[id]=Cert(n,c,g,d);
 }

 function get(uint id) public view returns(string memory,string memory,string memory,string memory){
  Cert memory x=certs[id];
  return(x.name,x.course,x.grade,x.date);
 }
}

EX 06 – Clinical Trial

Procedure: Deploy contract → Enroll → Record data → Retrieve.

pragma solidity ^0.8.0;
contract ClinicalTrial{
 struct P{string name;string data;}
 mapping(uint=>P) public p;

 function enroll(uint id,string memory n) public{p[id].name=n;}
 function record(uint id,string memory d) public{p[id].data=d;}
 function get(uint id) public view returns(string memory,string memory){
  return(p[id].name,p[id].data);
 }
}

EX 07 – Land Registration

Procedure: Deploy → Register land → Query details.

pragma solidity ^0.8.0;
contract Land{
 struct L{string owner;string loc;}
 mapping(uint=>L) public lands;

 function register(uint id,string memory o,string memory l) public{
  lands[id]=L(o,l);
 }

 function get(uint id) public view returns(string memory,string memory){
  return(lands[id].owner,lands[id].loc);
 }
}

EX 08 – Smart Toll

Procedure: Register → Pay toll → Check balance.

pragma solidity ^0.8.0;
contract Toll{
 mapping(address=>uint) public bal;

 function register() public{bal[msg.sender]=100;}
 function pay(uint a) public{require(bal[msg.sender]>=a);bal[msg.sender]-=a;}
 function get() public view returns(uint){return bal[msg.sender];}
}
